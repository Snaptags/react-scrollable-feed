```bash
npm install --save react-scrollable-feed
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'react-scrollable-feed'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
